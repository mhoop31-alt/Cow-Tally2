from kivy.app import App from kivy.uix.boxlayout import BoxLayout from kivy.uix.label import Label from kivy.uix.button import Button from kivy.uix.popup import Popup from kivy.graphics import Color, Line from kivy.properties import NumericProperty class LivestockCounter(App): cow_count = NumericProperty(0) calf_count = NumericProperty(0) def build(self): main_layout = BoxLayout(orientation='vertical', padding=10, spacing=10) # --- COW SECTION --- cow_box = BoxLayout(orientation='vertical', spacing=5, size_hint_y=0.38) cow_box.add_widget(Label(text="COWS", color=(1, 0.84, 0, 1), font_size='45sp', bold=True)) self.cow_label = Label(text="0", font_size='90sp', bold=True) cow_box.add_widget(self.cow_label) btn_add_cow = Button(text="+ ADD COW", background_color=(0.15, 0.65, 0.27, 1), font_size='35sp', bold=True) btn_add_cow.bind(on_release=lambda x: self.update_count('cow', 1)) cow_box.add_widget(btn_add_cow) cow_controls = BoxLayout(spacing=10, size_hint_y=0.4) btn_minus_cow = Button(text="- 1", background_color=(0.3, 0.3, 0.3, 1), font_size='25sp', bold=True) btn_minus_cow.bind(on_release=lambda x: self.update_count('cow', -1)) btn_clr_cow = Button(text="CLR", background_color=(0.3, 0.3, 0.3, 1), font_size='25sp', bold=True) btn_clr_cow.bind(on_release=lambda x: self.confirm_reset('cow')) cow_controls.add_widget(btn_minus_cow) cow_controls.add_widget(btn_clr_cow) cow_box.add_widget(cow_controls) main_layout.add_widget(cow_box) # --- CALF SECTION --- calf_box = BoxLayout(orientation='vertical', spacing=5, size_hint_y=0.38) calf_box.add_widget(Label(text="CALVES", color=(1, 0.84, 0, 1), font_size='45sp', bold=True)) self.calf_label = Label(text="0", font_size='90sp', bold=True) calf_box.add_widget(self.calf_label) btn_add_calf = Button(text="+ ADD CALF", background_color=(0, 0.48, 1, 1), font_size='35sp', bold=True) btn_add_calf.bind(on_release=lambda x: self.update_count('calf', 1)) calf_box.add_widget(btn_add_calf) calf_controls = BoxLayout(spacing=10, size_hint_y=0.4) btn_minus_calf = Button(text="- 1", background_color=(0.3, 0.3, 0.3, 1), font_size='25sp', bold=True) btn_minus_calf.bind(on_release=lambda x: self.update_count('calf', -1)) btn_clr_calf = Button(text="CLR", background_color=(0.3, 0.3, 0.3, 1), font_size='25sp', bold=True) btn_clr_calf.bind(on_release=lambda x: self.confirm_reset('calf')) calf_controls.add_widget(btn_minus_calf) calf_controls.add_widget(btn_clr_calf) calf_box.add_widget(calf_controls) main_layout.add_widget(calf_box) # --- THE 3-ROW TOTAL BOX --- self.total_box = BoxLayout(orientation='vertical', padding=10, spacing=2, size_hint_y=0.24) with self.total_box.canvas.before: Color(1, 0.84, 0, 1) self.line = Line(width=2, rounded_rectangle=(0, 0, 0, 0, 10)) self.total_box.bind(pos=self.update_rect, size=self.update_rect) self.res_cow = Label(text="Cow Tally: 0", font_size='26sp', color=(0.15, 0.65, 0.27, 1), bold=True) self.res_calf = Label(text="Calf Tally: 0", font_size='26sp', color=(0, 0.48, 1, 1), bold=True) self.res_total = Label(text="GRAND TOTAL: 0", font_size='40sp', bold=True, color=(1, 0.84, 0, 1)) self.total_box.add_widget(self.res_cow) self.total_box.add_widget(self.res_calf) self.total_box.add_widget(self.res_total) main_layout.add_widget(self.total_box) return main_layout def update_rect(self, instance, value): self.line.rounded_rectangle = (instance.x, instance.y, instance.width, instance.height, 10) def update_count(self, category, amount): if category == 'cow': self.cow_count = max(0, self.cow_count + amount) self.cow_label.text = str(self.cow_count) self.res_cow.text = f"Cow Tally: {self.cow_count}" else: self.calf_count = max(0, self.calf_count + amount) self.calf_label.text = str(self.calf_count) self.res_calf.text = f"Calf Tally: {self.calf_count}" self.res_total.text = f"GRAND TOTAL: {self.cow_count + self.calf_count}" def confirm_reset(self, category): # Creates a confirmation popup content = BoxLayout(orientation='vertical', padding=10, spacing=10) content.add_widget(Label(text=f"Clear all {category}s?", font_size='20sp')) btns = BoxLayout(spacing=10, size_hint_y=0.6) yes_btn = Button(text="YES", background_color=(0.8, 0, 0, 1), bold=True) no_btn = Button(text="NO", background_color=(0.3, 0.3, 0.3, 1), bold=True) btns.add_widget(yes_btn) btns.add_widget(no_btn) content.add_widget(btns) popup = Popup(title="Confirm Reset", content=content, size_hint=(0.8, 0.4)) yes_btn.bind(on_release=lambda x: [self.reset_count(category), popup.dismiss()]) no_btn.bind(on_release=popup.dismiss) popup.open() def reset_count(self, category): if category == 'cow': self.cow_count = 0 self.cow_label.text = "0" self.res_cow.text = "Cow Tally: 0" else: self.calf_count = 0 self.calf_label.text = "0" self.res_calf.text = "Calf Tally: 0" self.res_total.text = f"GRAND TOTAL: {self.cow_count + self.calf_count}" if __name__ == '__main__': LivestockCounter().run()
